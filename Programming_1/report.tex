\section*{Question E}
\begin{flushleft}
  $f(h) = L (0.5\pi r^2 - r^2\arcsin \frac{h}{r} - h\sqrt{r^2 - h^2}) - V$ \\
  $L=10 , r=1 , V=12.4$ \\
  二分法 ： 定义域为$[0,1]$ , 根为 $0.166166$。 \\
  牛顿法 ： 初值为$0$ , 根为 $0.166166$。\\
  割线法 ： 初值分别为$0$和$1$ , 根为 $0.166166$。\\
\end{flushleft}
